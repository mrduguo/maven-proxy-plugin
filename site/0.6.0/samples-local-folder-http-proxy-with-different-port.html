Samples: Configure http server listen port and hostname

This sample pom.xml set the http server listen on 8888. After start the proxy, try http://localhost:8888/basedir/pom.xml.

Plugin Configuration

<plugin>
        <groupId>org.duguo.maven.plugins</groupId>
        <artifactId>maven-proxy-plugin</artifactId>
        <version>0.6.0</version>
        <executions>
         <execution>
       <phase>validate</phase>
           <goals>
                 <goal>http</goal>
           </goals>
         </execution>
        </executions>
        <configuration>
            <port>8888</port>
            <hostname>localhost</hostname>
                <virtualRepositories>
                  <virtualRepository>
                          <requestPrefix>/basedir</requestPrefix>
                          <storageBase>${basedir}/target/testrepo</storageBase>
                          <downloadRepositories>
                                <downloadRepository>
                                  <id>local.basedir</id>
                                  <url>file://${basedir}</url>
                                </downloadRepository>
                          </downloadRepositories>
                  </virtualRepository>
                </virtualRepositories>
        </configuration>
</plugin>
